64  StatePtr addState(StatePtr state)
65  {
66  // we call setup here rather than use the setup event in case the state is added after
67  // setup event has occured
68  state->setSharedData(&sharedData);
69  state->setup();
70  ofAddListener(state->changeStateEvent, this, &ofxStateMachine::onChangeState);
71  states.insert(make_pair(state->getName(), state));

99  void changeState(const string& name)
100  {
101  stateIt it = states.find(name);
102  if (it == states.end()) ofLog(OF_LOG_ERROR, "No state with name: %s. Make sure you have added it to the state machine and you have set the state's name correctly. Set the name by implementing \"const string getName()\" in your state class", name.c_str());
103  else if (it->second != currentState)
104  {
105  if (currentState) currentState->stateExit();
106  currentState = it->second;
107  currentState->stateEnter();
108  }
109  }
